Sorts the collection of DftMapEntrys by Ft in descending order.

List<DftMapEntry> sortByFrequency() {
  final terms = List<DftMapEntry>.from(this);
  terms.sort((a, b) => b.dFt.compareTo(a.dFt));
  return terms;
}
